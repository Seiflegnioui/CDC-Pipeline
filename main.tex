% ── Diapositive : Qu'est-ce que le CDC ? ─────────────────────────
\begin{frame}{Qu'est-ce que le CDC~?}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \vspace{0.3cm}
      \begin{tcolorbox}[card, title={\faBookOpen\ Change Data Capture}]
        \textbf{Le CDC} (Change Data Capture) est une technique qui permet de
        \textbf{capturer automatiquement} chaque modification (insertion, mise à jour,
        suppression) survenue dans une base de données, puis de la \textbf{propager
        en temps réel} vers d'autres systèmes.

        \vspace{0.3cm}
        \textbf{Principe fondamental :}\\
        Au lieu de sonder la base périodiquement, on \textbf{écoute le journal
        de transactions} (oplog, WAL, binlog) pour détecter les changements
        dès qu'ils se produisent.
      \end{tcolorbox}
    \end{column}
    \begin{column}{0.42\textwidth}
      \begin{tikzpicture}[scale=0.8, every node/.style={scale=0.8}]
        % Base source
        \node[component=cMongo, minimum height=1.8cm] (src) {\faDatabase\\[0.3em] Base Source};
        % Journal
        \node[component=cDebezium, minimum height=1.2cm, below=0.8cm of src] (log) {\faFileAlt\\[0.2em] Journal (Oplog)};
        % Flèche d'écoute
        \draw[arrow, cDebezium] (src) -- node[right, font=\scriptsize, text=textsec] {écriture} (log);
        % CDC
        \node[component=accent, minimum height=1.2cm, right=1.2cm of log] (cdc) {\faPlug\\[0.2em] Moteur CDC};
        \draw[arrow, accent] (log) -- node[above, font=\scriptsize, text=textsec] {lecture} (cdc);
        % Destinations
        \node[component=cElastic, minimum height=1cm, above right=0.5cm and 1cm of cdc] (d1) {\faSearch\ Recherche};
        \node[component=cKafka, minimum height=1cm, below right=0.5cm and 1cm of cdc] (d2) {\faDatabase\ Analytics};
        \draw[arrow, cElastic] (cdc) -- (d1);
        \draw[arrow, cKafka] (cdc) -- (d2);
      \end{tikzpicture}
    \end{column}
  \end{columns}
\end{frame}

% ── Diapositive : Le problème résolu ──────────────────────────────
\begin{frame}{Le Problème que Nous Résolvons}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \begin{tcolorbox}[badcard, title={\faTimesCircle\ Sans CDC (Approche Batch)}]
        \begin{itemize}
            \item Jobs batch toutes les heures \faArrowRight\ résultats de recherche \textbf{obsolètes}
            \item Scripts de synchronisation manuels \faArrowRight\ \textbf{fragiles et sujets aux erreurs}
            \item Temps d'arrêt requis pour la réindexation complète
            \item Aucun journal d'audit des modifications
            \item \textbf{Latence :} plusieurs minutes à heures
        \end{itemize}
      \end{tcolorbox}
    \end{column}
    \begin{column}{0.48\textwidth}
      \begin{tcolorbox}[goodcard, title={\faCheckCircle\ Avec CDC (Temps Réel)}]
        \begin{itemize}
            \item Latence \textbf{inférieure à 200\,ms} de l'écriture à la recherche
            \item \textbf{Zéro modification} du code applicatif nécessaire
            \item Journal d'audit complet de chaque changement atomique
            \item Tolérant aux pannes~: survit aux redémarrages
            \item \textbf{Découplage total} source / destination
        \end{itemize}
      \end{tcolorbox}
    \end{column}
  \end{columns}

      \textcolor{accent}{\Huge \faBroadcastTower}
      \vspace{0.4cm}
      \begin{itemize}
        \item Se comporte comme un \textbf{réplica MongoDB}
        \item Lit les \textit{change streams} en temps réel
        \item \textbf{Mode snapshot~:} capture l'état initial au démarrage
        \item \textbf{SMT} (Single Message Transforms)~: nettoie et restructure les événements
        \item Gestion complète des \textbf{suppressions} (mode rewrite + tombstones)
      \end{itemize}
    \end{column}
